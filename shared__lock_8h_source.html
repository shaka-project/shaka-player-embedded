104 template <typename Mutex>
105 class shared_lock {
106  public:
107  using mutex_type = Mutex;
108 
109  shared_lock() : mutex_(nullptr), owns_lock_(false) {}
110  explicit shared_lock(mutex_type& mutex) : mutex_(&mutex) {
111  mutex_->lock_shared();
112  owns_lock_ = true;
113  }
114  shared_lock(shared_lock&& other)
115  : mutex_(other.mutex_), owns_lock_(other.owns_lock_) {
116  other.mutex_ = nullptr;
117  other.owns_lock_ = false;
118  }
119 
120  ~shared_lock() {
121  if (owns_lock_)
122  mutex_->unlock_shared();
123  }
124 
125  NON_COPYABLE_TYPE(shared_lock);

130  }

133  bool owns_lock() const { return owns_lock_; }
