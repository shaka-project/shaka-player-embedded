38 AsyncResults<void> JsManager::RunScript(const std::string& path) {
39  auto run_future = impl_->RunScript(path)->future();
40  // This creates a std::future that will invoke the given method when the
41  // wait()/get() method is called. This exists to convert the
42  // std::future<bool> that is returned into a
43  // std::future<variant<monostate, Error>> that AsyncResults expects.
44  // TODO: Find a better way to do this.
45  auto future = std::async(
46  std::launch ::deferred, [run_future]() -> variant<monostate, Error> {
47  if (run_future.get())
48  return monostate();
49 
50  return Error(ErrorType::NonShakaError, "Unknown script error");
51  });
52 
53  return future.share();
54 }
