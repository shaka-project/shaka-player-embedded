57 ExceptionOr<void> SourceBuffer::AppendBuffer(ByteBuffer data) {
58  if (!media_source_) {
59  return JsError::DOMException(
60  InvalidStateError, "SourceBuffer has been detached from the <video>.");
61  }
62  if (updating) {
63  return JsError::DOMException(InvalidStateError,
64  "Already performing an update.");
65  }
66 
67  if (media_source_->ready_state == MediaSourceReadyState::ENDED) {
68  media_source_->ready_state = MediaSourceReadyState::OPEN;
69  media_source_->ScheduleEvent<events::Event>(EventType::SourceOpen);
70  }
71 
72  using namespace std::placeholders; // NOLINT
73  append_buffer_ = std::move(data);
74  if (!media_source_->GetController()->AppendData(
75  type_, timestamp_offset_, append_window_start_, append_window_end_,
76  append_buffer_.data(), append_buffer_.size(),
77  std::bind(&SourceBuffer::OnAppendComplete, this, _1))) {
78  return JsError::DOMException(
79  InvalidStateError, "Unable to find source type " + to_string(type_));
80  }
81 
82  updating = true;
83  return {};
84 }

204 void SourceBuffer::OnAppendComplete(media::Status status) {
205  VLOG(1) << "Finish appending media segment: " << status;
206  updating = false;
207  if (status != media::Status::Success) {
208  Abort();
209  ScheduleEvent<events::Event>(EventType::Error);

211  ScheduleEvent<events::Event>(EventType::UpdateEnd);
