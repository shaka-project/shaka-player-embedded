100 void Promise::ResolveWith(Handle<JsValue> value, bool run_events) {
101  CHECK(CanResolve()) << "Can't reject JavaScript created Promises.";
102 #if defined(USING_V8)
103  (void)resolver_.handle()->Resolve(GetIsolate()->GetCurrentContext(), value);
104  // In V8, Promises are invoked automatically, but *after* executing some
105  // JavaScript. If we resolve it now, the handlers won't be invoked until we
106  // call into JavaScript again, and then only after that JavaScript runs. For
107  // example, if the next JavaScript is a timer, the timer will run first, then
108  // the Promise handlers will be invoked, which is not the correct order.
109  if (run_events)
110  GetIsolate()->RunMicrotasks();
111 #elif defined(USING_JSC)
112  LocalVar<JsValue> except;
113  if (!InvokeMethod(resolve_.handle(), Handle<JsObject>(), 1, &value,
114  &except)) {
115  OnUncaughtException(except, /* in_promise */ false);
116  }
117 #endif
118 }

121  CHECK(CanResolve()) << "Can't reject JavaScript created Promises.";
122 #if defined(USING_V8)
123  (void)resolver_.handle()->Reject(GetIsolate()->GetCurrentContext(),
124  error.error());
125  // See comment in ResolveWith().
126  if (run_events)

132  &except)) {
