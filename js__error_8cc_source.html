99 JsError JsError::DOMException(ExceptionCode code) {
100  // Careful here. We are creating a new object but we won't hold a reference
101  // to it. We are running on the event thread, so a GC run cannot happen yet.
102  // We will throw the wrapper which will keep the object alive.
103  RefPtr<dom::DOMException> except = new dom::DOMException(code);
104  except->stack = GetJsStack();
105  return JsError(except->JsThis());
106 }
