63 ExceptionOr<RefPtr<MediaKeySession>> MediaKeys::CreateSession(
64  optional<MediaKeySessionType> session_type) {
65  const MediaKeySessionType type =
66  session_type.value_or(MediaKeySessionType::Temporary);
67  if (!factory_->SupportsSessionType(type)) {
68  return JsError::DOMException(NotSupportedError,
69  "The given session type is not supported.");
70  }
71 
72  std::unique_lock<std::mutex> lock(mutex_);
73  RefPtr<MediaKeySession> session =
74  new MediaKeySession(type, factory_, implementation_, &helper_);
75  sessions_.emplace_back(session);
