110 void TaskRunner::Run(std::function<void(RunLoop)> wrapper) {
111  wrapper([this]() {
112  while (running_) {
113  // Handle a task. This will only handle one task, then loop.
114  if (HandleTask())
115  continue;
116 
117  if (!HasPendingWork()) {
118  waiting_.SignalAllIfNotSet();
119  }
120 
121  // We don't have any work to do, wait for a while.
122  OnIdle();
123  }
124 
125  // If we stop early, delete any pending tasks. This must be done on the
126  // worker thread so we can delete JavaScript objects.
127  tasks_.clear();
128  waiting_.SignalAllIfNotSet();
129  });
130 }

138 bool TaskRunner::HandleTask() {
139  // We need to be careful here because:
140  // 1) We may be called from another thread to change tasks.
141  // 2) The callback may change tasks (including its own).
142 
143  const uint64_t now = util::Clock::Instance.GetMonotonicTime();
144  impl::PendingTaskBase* task = nullptr;
145  {
146  // Find the earliest timer we can finish. If there are multiple with the
147  // same time, pick the one registered earlier (lower ID).
148  std::unique_lock<Mutex> lock(mutex_);
149  uint64_t min_time = std::numeric_limits<uint64_t>::max();
150  TaskPriority max_priority = TaskPriority::Timer;
151  for (auto it = tasks_.begin(); it != tasks_.end();) {
152  if ((*it)->should_remove) {
153  it = tasks_.erase(it);
154  } else {
155  if ((*it)->priority > max_priority) {
156  max_priority = (*it)->priority;
157  task = it->get();
158  } else if (max_priority == TaskPriority::Timer) {
159  const uint64_t it_time = (*it)->start_ms + (*it)->delay_ms;
160  if (it_time < now && it_time < min_time) {
161  min_time = it_time;
162  task = it->get();
163  }
164  }
165  ++it;
166  }
167  }
168  }
169 
170  if (!task)
171  return false;
172 
173 #ifdef USING_V8
174  if (!is_worker_) {
175  // V8 attaches v8::Local<T> instances to the most recent v8::HandleScope
176  // instance. By having a scope here, the task can create local handles and
177  // they will be freed after the task finishes.
178  v8::HandleScope handles(GetIsolate());
179  task->Call();
180  } else {
181  task->Call();
182  }
183 #else
184  // Other JavaScript engines use ref-counting for local handles.
185  task->Call();
186  (void)is_worker_;
187 #endif
188 
189  if (task->loop) {
190  task->start_ms = now;
191  } else {
192  task->should_remove = true;
193  // Will be removed in the next iteration.
194  }
195  return true;
196 }
