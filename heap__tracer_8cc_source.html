43 void HeapTracer::Trace(const Traceable* ptr) {
44  std::unordered_set<const Traceable*> to_trace;
45  {
46  std::unique_lock<Mutex> lock(mutex_);
47  to_trace = std::move(pending_);
48  to_trace.insert(ptr);
49  DCHECK(pending_.empty());
50 
51  // We need to be careful about circular dependencies. It should not
52  // happen, but we do not want to get into an infinite loop. Only
53  // traverse if we have not seen it before.
54  for (auto it = to_trace.begin(); it != to_trace.end();) {
55  if (*it && alive_.count(*it) == 0) {
56  alive_.insert(*it);
57  ++it;
58  } else {
59  it = to_trace.erase(it);
60  }
61  }
62  }
63 
64  for (const Traceable* item : to_trace)
65  item->Trace(this);
66 }
