118 struct hash<shaka::ByteBuffer> {
119  size_t operator()(const shaka::ByteBuffer& buffer) const {
120  // Introduce some noise to make small buffers have a more distributed hash.
121  uint64_t ret = 0xacbdcfd0e1f20304;
122 
123  const uint8_t* ptr = buffer.data();
124  for (size_t count = buffer.size(); count > 0; ptr++, count--) {
125  // Rotate the number to make the order of bytes matter and so we affect
126  // the whole number.
127  ret = (ret << 8) | (ret >> 56);
128 
129  // "insert" the data into the hash.
130  ret ^= *ptr;
131  }
132 
133  // Intentionally truncate on 32-bit platforms.
134  return static_cast<size_t>(ret);
135  }
136 };
