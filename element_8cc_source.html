57 optional<std::string> Element::TextContent() const {
58  std::string ret;
59  for (auto& child : child_nodes()) {
60  if (child->node_type() == TEXT_NODE) {
61  ret.append(static_cast<Text*>(child.get())->data());
62  } else if (child->is_element()) {
63  auto temp = child->TextContent();
64  DCHECK(temp.has_value());
65  ret.append(temp.value());
66  }
67  }

103 void Element::SetAttributeNS(const std::string& ns, const std::string& key,
104  const std::string& value) {
105  // |key| is a qualified name, e.g. 'foo:bar'. We search by local name, so
106  // split to get just 'bar'.
107  auto split_at = key.find(':');
108  CHECK_NE(split_at, std::string::npos);
109  const std::string local_name = key.substr(split_at + 1);
110  const std::string prefix = key.substr(0, split_at);
111 
112  auto it = FindAttributeNS(ns, local_name);
113  if (it != attributes_.end())
114  (*it)->value = value;
115  else
116  attributes_.push_back(new Attr(this, local_name, ns, prefix, value));
117 }
