83 template <typename Func>
84 class PendingTask : public PendingTaskBase {
85  public:
86  static_assert(std::is_base_of<memory::Traceable,
87  typename std::decay<Func>::type>::value,
88  "Traceable callback object must be Traceable");
89  using Ret = typename std::result_of<Func()>::type;
90 
91  PendingTask(Func&& callback, const std::string& name, TaskPriority priority,
92  uint64_t delay_ms, int id, bool loop)
93  : PendingTaskBase(priority, delay_ms, id, loop),
94  callback(std::forward<Func>(callback)),
95  event(new ThreadEvent<Ret>(name)) {}
96 
97  void Call() override {
98  // If this were C++17, we could use if-constexpr:
99  //
100  // if constexpr (std::is_same<Ret, void>::value) {
101  // callback();
102  // event->SignalAllIfNotSet();
103  // } else {
104  // event->SignalAllIfNotSet(callback());
105  // }
106 
107  SetHelper<Func, Ret>::Set(&callback, &event);
108  }
109 
110  void Trace(memory::HeapTracer* tracer) const override {
111  tracer->Trace(&callback);
112  }
113 
114  typename std::decay<Func>::type callback;
115  std::shared_ptr<ThreadEvent<Ret>> event;
116 
117  private:
118  template <typename F, typename R>
119  struct SetHelper {
120  static void Set(typename std::decay<F>::type* callback,
121  std::shared_ptr<ThreadEvent<R>>* event) {
122  (*event)->SignalAllIfNotSet((*callback)());
123  }
124  };
125  template <typename F>
126  struct SetHelper<F, void> {
127  static void Set(typename std::decay<F>::type* callback,
128  std::shared_ptr<ThreadEvent<void>>* event) {
129  (*callback)();
130  (*event)->SignalAllIfNotSet();
131  }
132  };
133 };

250  template <typename Func>
251  std::shared_ptr<ThreadEvent<impl::RetOf<Func>>> AddInternalTask(
252  TaskPriority priority, const std::string& name, Func&& callback) {
253  DCHECK(priority != TaskPriority::Timer) << "Use AddTimer for timers";
254 
255  std::unique_lock<Mutex> lock(mutex_);
256  const int id = ++next_id_;
257  auto pending_task = new impl::PendingTask<Func>(
258  std::forward<Func>(callback), name, priority, 0, id, /* loop */ false);
259  tasks_.emplace_back(pending_task);
260  pending_task->event->SetProvider(&worker_);
261 
262  return pending_task->event;
263  }
