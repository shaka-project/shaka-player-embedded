138 void SetGenericPropertyRaw(Handle<JsObject> object, const std::string& name,
139  Handle<JsFunction> getter,
140  Handle<JsFunction> setter) {
141  // TODO: Find a better way to do this.
142  // This works by effectively running the following JavaScript:
143  // Object.defineProperty($object, $name, {get: $getter, set: $setter});
144  LocalVar<JsValue> js_Object =
145  GetMemberRaw(JSContextGetGlobalObject(GetContext()), "Object");
146  CHECK(js_Object && IsObject(js_Object));
147  LocalVar<JsValue> define_prop =
148  GetMemberRaw(UnsafeJsCast<JsObject>(js_Object), "defineProperty");
149  CHECK(define_prop && GetValueType(define_prop) == JSValueType::Function);
150 
151  LocalVar<JsObject> props(CreateObject());
152  SetMemberRaw(props, "get", getter);
153  if (setter)
154  SetMemberRaw(props, "set", setter);
155 
156  LocalVar<JsValue> args[] = {object, ToJsValue(name), props};
157  LocalVar<JsValue> except;
158  CHECK(InvokeMethod(UnsafeJsCast<JsObject>(define_prop),
159  UnsafeJsCast<JsObject>(js_Object), 3, args, &except))
160  << ConvertToString(except);
161 }

163 bool InvokeConstructor(Handle<JsFunction> ctor, int argc,
164  LocalVar<JsValue>* argv,
165  LocalVar<JsValue>* result_or_except) {
166  JSValueRef except = nullptr;
167  std::vector<JSValueRef> args(argv, argv + argc);
168  LocalVar<JsValue> ret =
169  JSObjectCallAsConstructor(GetContext(), ctor, argc, args.data(), &except);
170  if (except) {
171  *result_or_except = except;
172  return false;
173  } else {
174  *result_or_except = ret;
175  return true;
176  }

195 std::string ConvertToString(Handle<JsValue> value) {
196  LocalVar<JsString> str(JSValueToStringCopy(GetContext(), value, nullptr));
197  if (!str)
198  return "";
199 
200  const size_t max_size = JSStringGetMaximumUTF8CStringSize(str);
201  std::string ret(max_size, '\0');
202  const size_t real_size = JSStringGetUTF8CString(str, &ret[0], ret.size());
203  ret.resize(real_size - 1); // Subtract 1 to remove the null-terminator.
204  return ret;
205 }

326 JSValueType GetValueType(Handle<JsValue> value) {
327  JSContextRef cx = GetContext();
328  switch (JSValueGetType(cx, value)) {
329  case kJSTypeUndefined:
330  return JSValueType::Undefined;
331  case kJSTypeNull:
332  return JSValueType::Null;
333  case kJSTypeBoolean:
334  return JSValueType::Boolean;
335  case kJSTypeNumber:
336  return JSValueType::Number;
337  case kJSTypeString:
338  return JSValueType::String;
339  case kJSTypeObject:
340  break;
341  default:
342  LOG(FATAL) << "Unknown JavaScript value type";
343  }
344  // Note JSC doesn't support symbols.
345 
346  if (JSValueIsArray(cx, value))
347  return JSValueType::Array;
348  if (JSObjectIsFunction(cx, JSValueToObject(cx, value, nullptr)))
349  return JSValueType::Function;
350 
351  switch (JSValueGetTypedArrayType(cx, value, nullptr)) {
352  case kJSTypedArrayTypeArrayBuffer:
353  return JSValueType::ArrayBuffer;
354  case kJSTypedArrayTypeFloat32Array:
355  return JSValueType::Float32Array;
356  case kJSTypedArrayTypeFloat64Array:
357  return JSValueType::Float64Array;
358  case kJSTypedArrayTypeInt16Array:
359  return JSValueType::Int16Array;
360  case kJSTypedArrayTypeInt32Array:
361  return JSValueType::Int32Array;
362  case kJSTypedArrayTypeInt8Array:
363  return JSValueType::Int8Array;
364  case kJSTypedArrayTypeUint16Array:
365  return JSValueType::Uint16Array;
366  case kJSTypedArrayTypeUint32Array:
367  return JSValueType::Uint32Array;
368  case kJSTypedArrayTypeUint8Array:
369  return JSValueType::Uint8Array;
370  case kJSTypedArrayTypeUint8ClampedArray:
371  return JSValueType::Uint8ClampedArray;
372 
373  case kJSTypedArrayTypeNone:
374  break;
375  default:
376  LOG(FATAL) << "Unknown JavaScript TypedArray type";
377  }
378 
379  if (IsInstanceOfStandardType(value, "Boolean"))
380  return JSValueType::BooleanObject;
381  if (IsInstanceOfStandardType(value, "String"))
382  return JSValueType::StringObject;
383  if (IsInstanceOfStandardType(value, "Number"))
384  return JSValueType::NumberObject;
385  if (IsInstanceOfStandardType(value, "Promise"))
386  return JSValueType::Promise;
387 
388  return JSValueType::OtherObject;
389 }
