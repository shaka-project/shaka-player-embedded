197 // Note that "no value" < "has value" for any value.
198 // See https://en.cppreference.com/w/cpp/utility/optional/operator_cmp
199 template <class A, class B>
200 bool operator==(const optional<A>& lhs, const optional<B>& rhs) {
201  if (!lhs.has_value() || !rhs.has_value())
202  return lhs.has_value() == rhs.has_value();
203  else
204  return lhs.value() == rhs.value();
205 }
